01660 protected:
01661     // -----------------------------------------------------------------------
01662     //  Protected data members
01663     //
01664     //  fCurrentNode
01665     //  fCurrentParent
01666     //      Used to track the current node during nested element events. Since
01667     //      the tree must be built from a set of disjoint callbacks, we need
01668     //      these to keep up with where we currently are.
01669     //
01670     //  fCurrentEntity
01671     //      Used to track the current entity decl.  If a text decl is seen later on,
01672     //      it is used to update the encoding and version information.
01673     //
01674     //  fDocument
01675     //      The root document object, filled with the document contents.
01676     //
01677     //  fCreateEntityReferenceNodes
01678     //      Indicates whether entity reference nodes should be created.
01679     //
01680     //  fIncludeIgnorableWhitespace
01681     //      Indicates whether ignorable whiltespace should be added to
01682     //      the DOM tree for validating parsers.
01683     //
01684     //  fScanner
01685     //      The scanner used for this parser. This is created during the
01686     //      constructor.
01687     //
01688     //  fImplementationFeatures
01689     //      The implementation features that we use to get an implementation
01690     //      for use in creating the DOMDocument used during parse. If this is
01691     //      null then the default DOMImplementation is used
01692     //
01693     //  fNodeStack
01694     //      Used to track previous parent nodes during nested element events.
01695     //
01696     //  fParseInProgress
01697     //      Used to prevent multiple entrance to the parser while its doing
01698     //      a parse.
01699     //
01700     //  fWithinElement
01701     //      A flag to indicate that the parser is within at least one level
01702     //      of element processing.
01703     //
01704     //  fDocumentType
01705     //      Used to store and update the documentType variable information
01706     //      in fDocument
01707     //
01708     //  fDocumentVector
01709     //      Store all the previous fDocument(s) (thus not the current fDocument)
01710     //      created in this parser.  It is destroyed when the parser is destructed.
01711     //
01712     //  fCreateCommentNodes
01713     //      Indicates whether comment nodes should be created.
01714     //
01715     //  fDocumentAdoptedByUser
01716     //      The DOMDocument ownership has been transferred to application
01717     //      If set to true, the parser does not own the document anymore
01718     //      and thus will not release its memory.
01719     //
01720     //  fInternalSubset
01721     //      Buffer for storing the internal subset information.
01722     //      Once complete (after DOCTYPE is finished scanning), send
01723     //      it to DocumentType Node
01724     //
01725     //   fGrammarPool
01726     //      The grammar pool passed from external application (through derivatives).
01727     //      which could be 0, not owned.
01728     //
01729     //  fCreateSchemaInfo
01730     //      Indicates whether element and attributes will have schema info associated
01731     //
01732     // -----------------------------------------------------------------------
01733     bool                          fCreateEntityReferenceNodes;
01734     bool                          fIncludeIgnorableWhitespace;
01735     bool                          fWithinElement;
01736     bool                          fParseInProgress;
01737     bool                          fCreateCommentNodes;
01738     bool                          fDocumentAdoptedByUser;
01739     bool                          fCreateSchemaInfo;
01740     XMLScanner*                   fScanner;
01741     XMLCh*                        fImplementationFeatures;
