00621 inline const XMLCh* XMLUri::getUriText() const
00622 {
00623     //
00624     //  Fault it in if not already. Since this is a const method and we
00625     //  can't use mutable members due the compilers we have to support,
00626     //  we have to cast off the constness.
00627     //
00628     if (!fURIText)
00629         ((XMLUri*)this)->buildFullText();
00630 
00631     return fURIText;
00632 }
