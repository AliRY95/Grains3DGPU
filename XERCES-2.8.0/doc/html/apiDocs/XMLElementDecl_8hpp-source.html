00512     // -----------------------------------------------------------------------
00513     //  Data members
00514     //
00515     //  fElementName
00516     //      This is the name of the element decl.
00517     //
00518     //  fCreateReason
00519     //      We sometimes have to put an element decl object into the elem
00520     //      decl pool before the element's declaration is seen, such as when
00521     //      its used in another element's content model or an att list is
00522     //      seen for it. This flag tells us whether its been declared, and
00523     //      if not why it had to be created.
00524     //
00525     //  fId
00526     //      The unique id of this element. This is created by the derived
00527     //      class, or more accurately the grammar that owns the objects
00528     //      of the derived types. But, since they all have to have them, we
00529     //      let them all store the id here. It is defaulted to have the
00530     //      value fgInvalidElem until explicitly set.
00531     //
00532     //  fExternalElement
00533     //      This flag indicates whether or the element was declared externally.
00534     // -----------------------------------------------------------------------

00538     unsigned int        fId;
00539     bool                fExternalElement;
